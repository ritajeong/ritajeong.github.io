package boj;

import java.util.Scanner;

public class b1074Z {
	static int n, r, c, len,start;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		r = sc.nextInt();
		c = sc.nextInt();
		double length = Math.pow(2, n);
		len = (int) length;
		len /= 2;
		func(r,c,len);
		sc.close();
	}

	static void func(int r, int c, int len) {
		if (len == 1) {
			if(r==0 &&c==0)
				System.out.println(start);
			else if(r==0&&c==1)
				System.out.println(start+1);
			else if(r==1&&c==0)
				System.out.println(start+2);
			else if(r==1&&c==1)
				System.out.println(start+3);
			return;
		}
		if (r < len && c < len) {
			start += len*0;
			func(r, c, len/2);
		} else if (r < len && c >= len) {
			start += len*len;
			func(r, c - len, len/2);
		} else if (r >= len && c < len) {
			start += len*len*2;
			func(r - len, c, len/2);
		} else{  //(r >= len && c >= len
			start += len*len*3;
			func(r - len, c - len, len/2);
		}
	}
}
