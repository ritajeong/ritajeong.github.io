#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
bool comp(const pair<int, int>  &a, const pair<int, int> &b) {
	return a.second < b.second;
}
int main() {
	int n;
	cin >> n;
	vector<pair<int, int>> v(n);
	for (int i = 0; i < n; i++) {
		cin >> v[i].first;
		cin >> v[i].second;
	}
	sort(v.begin(), v.end());
	sort(v.begin(), v.end(), comp);
	int min = v[0].second;
	int cnt = 0;
	cnt++;

	for (int i = 1; i < n; i++) {
		if (v[i].first >= min) {
			min = v[i].second;
			cnt++;
		}
	}
	cout << cnt;
}
