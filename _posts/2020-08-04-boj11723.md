```cpp
//이전에 제출한 코드에서 코드반복 줄임. 
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
vector<int> v;
int check(int n) {
	vector<int>::iterator iter;
	iter = find(v.begin(), v.end(), n);
	if (iter == v.end()) //없음
		return 0;
	else
		return 1;
}
void remove(int n) { //원소가 있는지 사전에 체크한 후 실행
	vector<int>::iterator iter;
	iter = find(v.begin(), v.end(), n);
	v.erase(iter);
} 
int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int m, n;
	cin >> m;
	string str;
	while (m--) {
		cin >> str;
		if (str == "check") {
			cin >> n;
			cout << check(n) << '\n';
		}
		else if (str == "add")
		{
			cin >> n;
			if (!check(n)) v.push_back(n);
		}
		else if (str == "remove")
		{
			cin >> n;
			if (check(n)) remove(n);
		}
		else if (str == "toggle")
		{
			cin >> n;
			if (check(n)) remove(n);
			else v.push_back(n);
		}
		else if (str == "empty")
			v.clear();
		else if (str == "all") {
			v.clear();
			for (int i = 0; i < 20; i++)
				v.push_back(i + 1);
		}
	}
}
```
